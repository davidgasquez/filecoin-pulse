```sql years
select
  distinct extract(year from date) as year,
from daily_metrics
```

```sql daily_metrics
select
  date,
  onboarded_data_tibs,
  deals,
  unique_clients,
  unique_providers
from daily_metrics
where extract(year from date) like '${inputs.year}'
order by date desc
```

## Top Providers

```sql providers
select
  provider_id,
  '/provider/' || provider_id as link,
  sum(onboarded_data_tibs) as onboarded_data_tibs
from provider_metrics
where extract(year from date) like '${inputs.year}'
group by provider_id, link
order by 3 desc
limit 200
```
